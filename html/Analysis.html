def anagram(text)
  head, tail = [], text.split('')
  stack = [[head, tail]]
  result = []

  while stack.size > 0
    head, tail = stack.pop
    if tail.size.zero?
      result << head
    else
      tail.each_with_index do |_, i|
        _tail = tail.dup
        curr  = _tail.delete_at(i)
        _head = head.dup
        _head << curr
        stack << [_head, _tail]
      end
    end
  end

  result.map! { |p| p.join('') }
  result.uniq!
  result
end

def factor(n)
  factors = []
  (1..n).each do |x|
    factors << x if n % x == 0
  end

  factors
end

def frequency(text)
  text  = text.downcase.gsub(/\s*/, '')
  chars = text.split('')
  freqs = Hash[('a'..'z').to_a.zip([0] * 26)]

  chars.each do |c|
    freqs[c] += 1
  end

  freqs
end

def ioc(text)
  c     = 26.0
  freqs = frequency(text)
  n     = text.length

  f = freqs.values.reduce(0) { |a, e| a + (e * (e - 1)) }
  f / ((n * (n - 1)) / c)
end

def vowels(text)
  vowels = %w(A E I O U)
  chars = text.upcase.split('')
  vowel_count = 0
  chars.each do |c|
    vowel_count += 1 if vowels.include? c
  end

  vowel_count.to_f / text.length
end
